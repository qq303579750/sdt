seajs.use(["$", "sticky"], function($, sticky) {
    // sticky
    var st = sticky("#stick", 30, function(status) {
        if (status) {
            window.console && console.log("stick");
        } else {
            window.console && console.log("unstick");
        }
    });

    // fixed
    $('<div id="nav">顶层fixed导航</div>').appendTo("body");
    sticky.fix("#nav");
});
